import { Component } from '@angular/core';
import { Observable } from 'rxjs';

@Component({
  selector: 'app-demo',
  templateUrl: './demo.component.html',
  styleUrls: ['./demo.component.css']
})
export class DemoComponent
{
  getData_click()
  {

    let observableObj = new Observable( (subscriber) => {
          subscriber.next("Data generated by observable -- 1");
          subscriber.next("Data generated by observable -- 2");

         let x = 6;
         if( x % 2 == 1) {
           subscriber.error("Error Occurred while processing observable");
         }

          setTimeout(
            () =>  {
              subscriber.next("Data generated by observable -- 3");
              subscriber.complete();
            }, 3000);  // wait the process for 3 sec
    });

    // subscribe or consuming observables
    observableObj.subscribe(
      (result:any) =>  { console.log(result);  },
      (error:any) =>  {    console.log(error);   },
      ()  => {  console.log("Send a email notification -- task is completed"); }
      );


     console.log("This is the last line in getData_click method") ;
  }
}
